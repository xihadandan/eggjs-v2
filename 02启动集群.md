### 集群启动流程
- wellapp-framework\node_modules\egg-cluster\index.js
```javascript
const Master = require('./lib/master');
/**
 * cluster start flow:
 *
 * [startCluster] -> master -> agent_worker -> new [Agent]       -> agentWorkerLoader
 *                         `-> app_worker   -> new [Application] -> appWorkerLoader
 *
 */
exports.startCluster = function(options, callback) {
  new Master(options).ready(callback); // 实例化主工人 
  // ready() 方法的声明是在get-ready\index.js中的ready(flagOrFunction) {}
};

```

### 一次性准备事件
- wellapp-framework\node_modules\egg-cluster\node_modules\get-ready\index.js
```javascript

class Ready {
  ...
  ready(flagOrFunction) {
    // register a callback
    if (flagOrFunction === undefined || is.function(flagOrFunction)) {
      return this.register(flagOrFunction);
    }
    // emit callbacks
    this.emit(flagOrFunction);
  }
  ...
  static mixin(obj) {
    if (!obj) return;
    const ready = new Ready();
    // delegate method
    obj.ready = flagOrFunction => ready.ready(flagOrFunction);
  }
}


function mixin(object) {
  Ready.mixin(object);
}

```

## 过程
- 实例化主进程
- 主进程构造函数中实例化“信使”
- “信使”构造函数中设置master（主进程）
- 主进程构造函数中检测端口得到集群端口`this.detectPorts`
- 通过childprocess 模块fork代理进程` const agentWorker = childprocess.fork(this.getAgentWorkerFile(), args, opt);`
- 主进程中监听代理进程发送的消息 `agentWorker.on('message', msg => {`
- 代理进程中向主进程发送 `process.send({ action: 'agent-start', to: 'master' });`
- 主进程中得到代理进程发送的消息并转发“将代理的消息转发给信使”
- “信使”实例调用 this.master.emit 发送给主进程
- 主进程构造函数中监听`this.on('agent-start', this.onAgentStart.bind(this));`

- detectPorts

- forkAppWorkers


### egg-ready
在接到 egg-ready 消息后方可发送消息。只有当 Master 确认所有 Agent 和 Worker 启动成功并就绪后，才会通过 messenger 发送 egg-ready 通知每个 Agent 和 Worker，表示一切就绪，可使用 IPC
